Умови та цикли


Умови (if, else) - це спосіб приймати рішення у програмі, виконуючи код лише тоді, коли виконується певна умова.
Цикли (for, while) дозволяють повторювати блоки коду кілька разів, поки виконується певна умова.

Умовні оператори if / else


int x = 15;

if (x > 10) {
    System.out.println("Більше 10");
} else {
    System.out.println("10 або менше");
}

else if


int score = 75;

if (score >= 90) {
    System.out.println("Відмінно");
} else if (score >= 70) {
    System.out.println("Добре");
} else {
    System.out.println("Спробуй ще");
}

switch


int day = 2;

switch (day) {
    case 1: System.out.println("Понеділок"); break;
    case 2: System.out.println("Вівторок"); break;
    default: System.out.println("Невідомий день");
}

Цикл for


for (int i = 1; i <= 5; i++) {
    System.out.println(i);
}

Цикл while


int i = 1;

while (i <= 5) {
    System.out.println(i);
    i++;
}

Цикл do while


int i = 1;

do {
    System.out.println(i);
    i++;
} while (i <= 5);

break та continue


for (int i = 1; i <= 5; i++) {
    if (i == 3) continue;
    System.out.println(i);
}
