Колекції та робота з даними


Колекції даних - це способи зберігати і групувати багато значень разом в одній змінній.Вони потрібні, щоб не створювати десятки окремих змінних для схожих даних.

ArrayList


ArrayList names = new ArrayList<>();
names.add("Anna");
names.add("Oleh");
System.out.println(names.get(0));

LinkedList


LinkedList numbers = new LinkedList<>();
numbers.add(5);
numbers.add(10);

HashSet


HashSet cities = new HashSet<>();
cities.add("Kyiv");
cities.add("Lviv");
cities.add("Kyiv");
System.out.println(cities);

HashMap


HashMap grades = new HashMap<>();
grades.put("Ivan", 10);
grades.put("Maria", 12);
System.out.println(grades.get("Ivan"));

Перебір колекцій


for (String name : names) {
    System.out.println(name);
}

Сортування


Collections.sort(names);
System.out.println(names);

Comparator


Collections.sort(names, Collections.reverseOrder());
